Sub AddAndRemoveRandomString()
    Dim lastRow As Long
    Dim i As Long
    Dim strArr() As String
    Dim randIndex As Long
    
    ' 추가할 문자열 목록
    strArr = Split("main^|^,list^|^,detail^|^,magnify^|^,add1^|^", ",")
    
    ' 마지막 행 찾기
    lastRow = Cells(Rows.Count, 61).End(xlUp).Row
    
    ' 각 행을 반복하면서 문자열 추가 및 삭제
    For i = 5 To lastRow
        ' 랜덤한 문자열 선택
        randIndex = Int((UBound(strArr) + 1) * Rnd())
        ' 기존 문자열과 랜덤 문자열 연결
        Cells(i, 61).Value = strArr(randIndex) & Cells(i, 61).Value
        ' 기존 문자열에서 ? 뒤의 문자열 삭제
        Cells(i, 61).Value = Split(Cells(i, 61).Value, "?")(0)
    Next i
End Sub

Sub ChangeTaxStatus()
    Dim lastRow As Long
    lastRow = Cells(Rows.Count, "AG").End(xlUp).Row ' AG 컬럼의 마지막 행 번호를 가져옴
    
    For i = 4 To lastRow ' 4번 행부터 마지막 행까지 반복
        If Cells(i, "AG").Value = "과세상품" Then
            Cells(i, "AG").Value = "t" ' 값을 "t"로 변경
        ElseIf Cells(i, "AG").Value = "면세상품" Then
            Cells(i, "AG").Value = "f" ' 값을 "f"로 변경
        End If
    Next i
End Sub
